
function swap(arr, indexa, indexb){
  let temp = arr[indexa]
  arr[indexa] = arr[indexb]
  arr[indexb] = temp
  return arr
}

let begintime = performance.now()
let isdone = false
while(!isdone){
  isdone = true
  for(let i=0;i<arr.length;i++){
    if(arr[i]>arr[i+1]){
      arr = swap(arr,i,i+1)
      isdone = false
    }
  }
}

console.log("manual sort: " + (performance.now()-begintime) + " ms")
